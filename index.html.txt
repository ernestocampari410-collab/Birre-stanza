<script>
const form = document.getElementById('signupForm');
const forum = document.getElementById('forum');
const beerBg = document.getElementById('beerBg');

// Funzione forum
form.addEventListener('submit', function(e) {
e.preventDefault();
const name = document.getElementById('name').value;
const message = document.getElementById('message').value;

const msgDiv = document.createElement('div');
msgDiv.classList.add('message');
msgDiv.innerHTML = `<strong>${name}:</strong> ${message}`;
forum.prepend(msgDiv);

form.reset();
});

// Genera birre sullo sfondo
const beerImages = [
'https://media.giphy.com/media/xT0xeJpnrWC4XWblEk/giphy.gif',
'https://media.giphy.com/media/l0HlQ7LRal0yKZ8y0/giphy.gif'
];

function createBeer() {
const img = document.createElement('img');
img.src = beerImages[Math.floor(Math.random() * beerImages.length)];
img.style.left = Math.random() * window.innerWidth + 'px';
img.style.animationDuration = (3 + Math.random() * 5) + 's';
beerBg.appendChild(img);

setTimeout(() => beerBg.removeChild(img), 8000);
}

setInterval(createBeer, 500);
